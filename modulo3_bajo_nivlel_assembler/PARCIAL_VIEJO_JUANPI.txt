ORG 1000H ; variables
NA db 23,24,25,26
FINTABLA DB ?
ORG 3000H ; SUBRUTINAS
SUBRUTINA: PUSH AX
PUSH BX
PUSHF
MOV BX, SP
ADD BX, 8
MOV BX, [BX]
MOV AH, [BX]
LOOP: CMP AH, 25
JS FIN
MOV [BX], AL 
FIN: POPF
POP BX
POP AX
RET
ORG 2000H  ; PROGRAMA PRINCIPAL
MOV AL, 25
MOV CX, OFFSET NA
MOV DL,  OFFSET FINTABLA - OFFSET NA
BUCLE: PUSH CX
CALL SUBRUTINA
POP CX
INC CX
DEC DL
CMP DL, 0
JZ FINAL
JMP BUCLE
FINAL: HLT
END
