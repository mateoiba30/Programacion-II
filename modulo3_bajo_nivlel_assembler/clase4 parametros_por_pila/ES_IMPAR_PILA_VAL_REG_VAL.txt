ORG 1000H
V1 DW 3

ORG 3000H
DIVIDIR:CMP BX,2
JC FIN;SI HAY ACARREO TERMINO
SUB BX,2
JMP DIVIDIR
FIN:RET

;SOLO ES NECESARIO MODIFICAR DX, NO IMPORTA SI MODIFICAMOS AX
ES_IMPAR:MOV BX,SP
ADD BX,2;BX APUNTA AL VALOR DE V1
MOV BX,[BX];BX TIENE EL VALOR DE V1

CALL DIVIDIR;SI EL RESTO (GUARDADO EN BX) ES CERO, ENTONCES ES PAR
CMP BX,0
JNZ FIN2
MOV DX,00H
FIN2:RET

ORG 2000H

MOV AX,V1
MOV DX,01H;POR REGISTROS SABEMOS QUE 01H PARA IMPAR, Y 00H PARA PAR
PUSH AX;POR AX PASAMOS EL NRO
CALL ES_IMPAR
POP AX

HLT
END
